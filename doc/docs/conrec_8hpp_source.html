53//=============================================================================
54//
55// CONREC is a contouring subroutine for rectangularily spaced data.
56//
57// It emits calls to a line drawing subroutine supplied by the user
58// which draws a contour map corresponding to real*4data on a randomly
59// spaced rectangular grid. The coordinates emitted are in the same
60// units given in the x() and y() arrays.
61//
62// Any number of contour levels may be specified but they must be
63// in order of increasing value.
64//
65// As this code is ported from FORTRAN-77, please be very careful of the
66// various indices like ilb,iub,jlb and jub, remeber that C/C++ indices
67// starts from zero (0)
68//
69//=============================================================================
70
71class Graphics;
72
73#include "pdetype.h"
74
79int conrec(PDEFIELD_TYPE **d,
80 int ilb,
81 int iub,
82 int jlb,
83 int jub,
84 double *x,
85 double *y,
86 int nc,
87 double *z,
88 Graphics *g,
89 int colour=1);
90// d ! matrix of data to contour
91// ilb,iub,jlb,jub ! index bounds of data matrix
92// x ! data matrix column coordinates
93// y ! data matrix row coordinates
94// nc ! number of contour levels
95// z ! contour levels in increasing order
96
97#endif
