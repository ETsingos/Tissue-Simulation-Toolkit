10/* Return a list of coordinates of pixels in range.
11 *
12 * The range is set by the **adhesion_zone_radius** parameter.
13 *
14 * Note that there's some overlap in functionality between this an the
15 * Neighbors class in adhesion_movement.cpp. They're kept separate however
16 * because that class is very performance-sensitive, and calculating the list
17 * dynamically as is done here will likely be quite a bit slower. That doesn't
18 * matter here, but it does there, so we have one fast and one flexible
19 * implementation.
20 */
21std::vector<PixelDisplacement> neighbor_list() {
22 std::vector<PixelDisplacement> result;
23 int radius = static_cast<int>(ceil(par.adhesion_zone_radius));
24 for (int dy = -radius; dy <= radius; ++dy) {
25 for (int dx = -radius; dx <= radius; ++dx) {
26 if (sqr(dy) + sqr(dx) <= sqr(par.adhesion_zone_radius))
27 result.emplace_back(dx, dy);
28 }
29 }
30 return result;
31}
