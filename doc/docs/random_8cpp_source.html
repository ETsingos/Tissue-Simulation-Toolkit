36double RANDOM(void) {
37/* Knuth's substractive method, see Numerical Recipes */
38 static int inext,inextp;
39 static long ma[56];
40 static int iff=0;
41 long mj,mk;
42 int i,ii,k;
43
44 if (idum < 0 || iff == 0) {
45 iff=1;
46 mj=labs(MSEED-labs(idum));
47 mj %= MBIG;
48 ma[55]=mj;
49 mk=1;
50 i=1;
51 do {
52 ii=(21*i) % 55;
53 ma[ii]=mk;
54 mk=mj-mk;
55 if (mk < MZ) mk += MBIG;
56 mj=ma[ii];
57 } while ( ++i <= 54 );
58 k=1;
59 do {
60 i=1;
61 do {
62 ma[i] -= ma[1+(i+30) % 55];
63 if (ma[i] < MZ) ma[i] += MBIG;
64 } while ( ++i <= 55 );
65 } while ( ++k <= 4 );
66 inext=0;
67 inextp=31;
68 idum=1;
69 }
70 if (++inext == 56) inext=1;
71 if (++inextp == 56) inextp=1;
72 mj=ma[inext]-ma[inextp];
73 if (mj < MZ) mj += MBIG;
74 ma[inext]=mj;
75 return mj*FAC;
76}

81long Seed(long seed) {

129long Randomize(void) {
130 // Set the seed according to the local time
131 int seed;
132
133 auto timepoint = std::chrono::system_clock::now();
134 auto since_epoch = timepoint.time_since_epoch();
135 auto seconds = std::chrono::duration_cast<std::chrono::seconds>(since_epoch);
136 auto milliseconds = std::chrono::duration_cast<std::chrono::milliseconds>(since_epoch);
137
138 seed=abs((int)((seconds.count()*milliseconds.count())%655337));
139 Seed(seed);
140 fprintf(stderr,"Random seed is %d\n",seed);
141 return seed;
142}

145double generateGaussianNoise(double mu, double sigma)
146{
147 static const double epsilon = std::numeric_limits<double>::min();
148 static const double two_pi = 2.0*3.14159265358979323846;
149
150 thread_local double z1;
151 thread_local bool generate;
152 generate = !generate;
153
154 if (!generate)
155 return z1 * sigma + mu;
156
157 double u1, u2;
158 do
159 {
160 u1 = RANDOM();
161 u2 = RANDOM();
162 }
163 while ( u1 <= epsilon );
164
165 double z0;
166 z0 = sqrt(-2.0 * log(u1)) * cos(two_pi * u2);
167 z1 = sqrt(-2.0 * log(u1)) * sin(two_pi * u2);
168 return z0 * sigma + mu;
169}
