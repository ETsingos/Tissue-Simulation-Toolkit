53template <typename T, size_t N> struct hash<const array<T, N>> {
54 typedef const array<T, N> argument_type;
55 typedef size_t result_type;
56 result_type operator()(const argument_type &a) const {
57 hash<T> hasher;

60 h = h * hasher(a[i]);
