416 // used internally by class CellularPotts
417 inline double AddSiteToMoments(int x,int y, double new_l=-1. ) {
418
419 // Add a site to the raw moments, then update and return the
420 // length of the cell
421
422 // sum_x, sum_y, sum_xx, sum_xy and sum_yy are adjusted
423 // Eventually this function may be used to carry
424 // out all necessary adminstration at once
425 sum_x+=x;
426 sum_y+=y;
427 sum_xx+=x*x;
428 sum_yy+=y*y;
429 sum_xy+=x*y;
430
431 // update length (see appendix. A, Zajac.jtb03), if length is not given
432 // NB. 24 NOV 2004. Found mistake in Zajac's paper. See remarks in
433 // method "Length(..)".
434
435 if (new_l<0.) {
436 length=Length(sum_x,sum_y,sum_xx,sum_yy,sum_xy,area);
437 } else {
438 length=new_l;
439 }
440 return length;
441 }

443 // used internally by class CellularPotts
444 inline double RemoveSiteFromMoments(int x,int y, double new_l=-1.) {
445
446 // Remove a site from the raw moments, then update and return the
447 // length of the cell
448
449 // sum_x, sum_y, sum_xx, sum_xy and sum_yy are adjusted
450 // Eventually this function may be used to carry
451 // out all necessary adminstration at once
452 sum_x-=x;
453 sum_y-=y;
454 sum_xx-=x*x;
455 sum_yy-=y*y;
456 sum_xy-=x*y;
457
458 // update length (see app. A, Zajac.jtb03), if length is not given
459 if (new_l<0.) {
460 length=Length(sum_x,sum_y,sum_xx,sum_yy,sum_xy,area);
461 } else {
462 length=new_l;
463 }
464 return length;
465 }

470 inline double Length(long int s_x,long int s_y,long int s_xx,
471 long int s_yy,long int s_xy,long int n) {
472 // prevent NaN when last pixel is deleted
473 if (n==0) {
474 return 0.;
475 }
476
477 // prevent NaN when last pixel is deleted
478 if (n==0) {
479 return 0.;
480 }
481
482 // inertia tensor (constructed from the raw momenta, see notebook)
483 double iyy=(double)s_xx-(double)s_x*s_x/(double)n;
484 double ixx=(double)s_yy-(double)s_y*s_y/(double)n;
485 double ixy=-(double)s_xy+(double)s_x*s_y/(double)n;
486
487 double rhs1=(ixx+iyy)/2., rhs2=sqrt( (ixx-iyy)*(ixx-iyy)+4*ixy*ixy )/2.;
488
489 double lambda_b=rhs1+rhs2;
490 //double lambda_a=rhs1-rhs2;
491
492 // according to Zajac et al. 2003:
493 //return 2*sqrt(lambda_b);
494 // Grumble, this is not right!!!
495 // Must divide by mass!!!!!!
496
497 // see: http://scienceworld.wolfram.com/physics/MomentofInertiaEllipse.html
498 // std::cerr << "n = " << n << "\n";
499 return 4*sqrt(lambda_b/n);
500
501 // 2*sqrt(lambda_b/n) give semimajor axis. We want the length.
502
503 }

506 inline void MajorMinorAxis(double *major_axis, double *minor_axis, double *v1, double *v2) {
507
508 // inertia tensor (constructed from the raw momenta, see notebook)
509 double iyy=(double)sum_xx-(double)sum_x*sum_x/(double)area;
510 double ixx=(double)sum_yy-(double)sum_y*sum_y/(double)area;
511 double ixy=-(double)sum_xy+(double)sum_x*sum_y/(double)area;
512
513 double rhs1=(ixx+iyy)/2., rhs2=sqrt( (ixx-iyy)*(ixx-iyy)+4*ixy*ixy )/2.;
514
515 double lambda_b=rhs1+rhs2;
516 double lambda_a=rhs1-rhs2;
517
518 // see: http://scienceworld.wolfram.com/physics/MomentofInertiaEllipse.html
519 // std::cerr << "n = " << n << "\n";
520 *major_axis=4*sqrt(lambda_b/area);
521 *minor_axis=4*sqrt(lambda_a/area);
522
523 if (ixy!=0) {
524 *v1=lambda_a-iyy;
525 *v2=ixy;
526 } else {
527 *v1=0;
528 *v2=1;
529 }
530
531 // normalize vector
532 double norm = sqrt((*v1) * (*v1) + (*v2) * (*v2));
533 *v1/=norm;
534 *v2/=norm;
535
536 // 2*sqrt(lambda_b/n) give semimajor axis. We want the length.
537 }
