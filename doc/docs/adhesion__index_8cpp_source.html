30double AttachedAngleCst::move_dh(ParPos from, ParPos to) const {
31 auto x_from = from - middle;
32 auto x_to = to - middle;
33 auto y = far - middle;
34
35 auto cos_theta_from = x_from.dot(y) / (x_from.length() * y.length());
36 if (cos_theta_from < -1.0) cos_theta_from = -1.0;
37 if (cos_theta_from > 1.0) cos_theta_from = 1.0;
38
39 auto cos_theta_to = x_to.dot(y) / (x_to.length() * y.length());
40 if (cos_theta_to < -1.0) cos_theta_to = -1.0;
41 if (cos_theta_to > 1.0) cos_theta_to = 1.0;
42
43 auto theta_from = acos(cos_theta_from);
44 auto theta_to = acos(cos_theta_to);
45
46 auto H_from = angle_cst_type.k * 0.5 * sqr((theta_from - angle_cst_type.t0));
47 auto H_to = angle_cst_type.k * 0.5 * sqr((theta_to - angle_cst_type.t0));
48
49 return H_to - H_from;
50}

80 // Map adhesion particles to bonds attached to them
81 std::unordered_map<ParId, std::vector<BondId>> make_bond_index(
82 ECMBoundaryState const & ecm_boundary)
83 {
84 std::unordered_map<ParId, std::vector<BondId>> bond_index;
85 for (auto const & id_bond : ecm_boundary.bonds) {
86 BondId bid = id_bond.first;
87 Bond const & bond = id_bond.second;
88
89 ParticleType p1_type = ecm_boundary.particles.at(bond.p1).type;
90 bool p1_adh = p1_type == ParticleType::adhesion;
91 bool p1_unfit = p1_adh || (p1_type == ParticleType::excluded);
92
93 ParticleType p2_type = ecm_boundary.particles.at(bond.p2).type;
94 bool p2_adh = p2_type == ParticleType::adhesion;
95 bool p2_unfit = p2_adh || (p2_type == ParticleType::excluded);
96
97 if (p1_adh && !p2_unfit)
98 bond_index[bond.p1].push_back(bid);
99
100 if (p2_adh && !p1_unfit)
101 bond_index[bond.p2].push_back(bid);
102 }
103 return bond_index;
104 }

134void AdhesionIndex::rebuild(ECMBoundaryState const & ecm_boundary) {
135 // Adhesion particles' positions are sent along by the other side,
136 // but the adhesion particles are part of our state, so they don't
137 // get to say where they are, we decided that. Unless they have
138 // created new adhesion particles for us and picked the location,
139 // which is weird but can happen with the original adhesion
140 // generation algorithm. So we save our existing adhesion
141 // particles' positions here, and keep them, only using the sent
142 // positions for adhesions particles we didn't have yet.
143 std::unordered_map<ParId, ParPos> adh_par_pos;
144 for (auto const & pixel_awes : adhesions_by_pixel_)
145 for (auto const & awe : pixel_awes.second)
146 adh_par_pos[awe.par_id] = awe.position;
147
148 auto bonds_for = make_bond_index(ecm_boundary);
149 auto angle_csts_for = make_angle_cst_index(ecm_boundary);
150
151 adhesions_by_pixel_.clear();
152 for (auto const id_par : ecm_boundary.particles) {
153 ParId pid = id_par.first;
154 Particle const & par = id_par.second;
155
156 if (par.type == ParticleType::adhesion) {
157 ParPos pos = adh_par_pos.count(pid) ? adh_par_pos[pid] : par.pos;
158 PixelPos containing_pixel(floor(pos.x), floor(pos.y));
159 adhesions_by_pixel_[containing_pixel].emplace_back(pid, pos);
160 auto & awe = adhesions_by_pixel_[containing_pixel].back();
161
162 for (BondId bid: bonds_for[pid]) {
163 auto const & bond = ecm_boundary.bonds.at(bid);
164
165 ParPos neighbor_pos;
166 if (bond.p1 == pid)
167 neighbor_pos = ecm_boundary.particles.at(bond.p2).pos;
168 else
169 neighbor_pos = ecm_boundary.particles.at(bond.p1).pos;
170
171 awe.bonds.emplace_back(
172 neighbor_pos, ecm_boundary.bond_types.at(bond.type));
173 }
174
175 for (AngleCstId aid: angle_csts_for[pid]) {
176 auto const & angle_cst = ecm_boundary.angle_csts.at(aid);
177
178 ParPos middle_pos = ecm_boundary.particles.at(angle_cst.p2).pos;
179
180 ParPos far_pos;
181 if (angle_cst.p1 == pid)
182 far_pos = ecm_boundary.particles.at(angle_cst.p3).pos;
183 else
184 far_pos = ecm_boundary.particles.at(angle_cst.p1).pos;
185
186 awe.angle_csts.emplace_back(
187 middle_pos, far_pos,
188 ecm_boundary.angle_cst_types.at(angle_cst.type));
189 }
190 }
191 }
192}

203void AdhesionIndex::move_adhesions(PixelPos from, PixelPos to) {
204 if (from == to) return;
205 auto it = adhesions_by_pixel_.find(from);
206 if (it != adhesions_by_pixel_.end()) {
207 for (auto & awe: it->second) {
208 awe.position += to - from;
209 ecm_interaction_tracker_.record_move_particle(awe.par_id, awe.position);
210 adhesions_by_pixel_[to].push_back(awe);
211 }
212 it->second.clear();
213 }
