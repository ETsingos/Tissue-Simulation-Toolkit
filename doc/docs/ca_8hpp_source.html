56 struct hash <const array<T, N> >
57 {
58 typedef const array<T, N> argument_type;
59 typedef size_t result_type;
60 result_type operator()(const argument_type& a) const
61 {
62 hash<T> hasher;
63 result_type h = 1;
64 for (result_type i = 0; i < N; ++i)
65 {
66 h = h * hasher(a[i]);
