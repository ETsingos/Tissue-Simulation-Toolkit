398  long int s_yy,long int s_xy,long int n) {
399 
400  // inertia tensor (constructed from the raw momenta, see notebook)
401  double iyy=(double)s_xx-(double)s_x*s_x/(double)n;
402  double ixx=(double)s_yy-(double)s_y*s_y/(double)n;
403  double ixy=-(double)s_xy+(double)s_x*s_y/(double)n;
